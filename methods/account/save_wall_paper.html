Example

const { TelegramClient } = require('gramjs');
const { functions, types } = require('gramjs/tl');

(async () => {
    const client = new TelegramClient(name, apiId, apiHash);
    await client.start();

    const result = await client.invoke(new functions.account.SaveWallPaperRequest(
        wallpaper=new types.InputWallPaper(
            id=-12398745604826,
            access_hash=-12398745604826
        ),
        unsave=False,
        settings=new types.WallPaperSettings()
    ));
    console.log(result);
})();
